辅助类：源字符串处理SourceFile

该类中更像一个结构体，将自己维护的两个成员变量data_,size_设置为共有，该类的两个构造函数会对源数据进行处理，以取消其中出现的反斜杠，实现比较简单，查询’/‘的函数为strchr(),源码如下：

class SourceFile
{
public:
    template
    //大小为N的数组的常量引用
    inline SourceFile(const char(&arr)[N])
        : data_(arr),
        size_(N - 1)
    {
        const char* slash = strrchr(data_, '/'); //用于查找在字符数组中最后一个出现'/'的位置，返回其引用
        if (slash)
        {
            data_ = slash + 1;
            size_ -= static_cast(data_ - arr);
        }
    }

    explicit SourceFile(const char* filename)
        : data_(filename)
    {
        const char* slash = strrchr(filename, '/');
        if (slash)
        {
            data_ = slash + 1;
        }
        size_ = static_cast(strlen(data_));
    }

    const char* data_;
    int size_;
};

// Efficient Integer to String Conversions, by Matthew Wilson.
template
size_ t convert(char buf[],T value){
    T i=value;
    char* p=buf;
    do{
        int lsd=static_cast(i%10);
        i/=10;
        *p++=zero[lsd];
    }while(i!=0);
    if(value<0){
        *p++='-';
    }
    *p='\0';
    std::reverse(buf,p);
    return p-buf;
}
  • size_t convertHex(char buf[],uintptr_t value);
size_t convertHex(char buf[], uintptr_t value)
{
    uintptr_t i = value;
    char* p = buf;

    do
    {
        int lsd = static_cast(i % 16);
        i /= 16;
        *p++ = digitsHex[lsd];
    } while (i != 0);

    *p = '\0';
    std::reverse(buf, p);

    return p - buf;
}
